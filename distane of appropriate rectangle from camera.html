from imutils import paths
import numpy as np
import imutils
from matplotlib import pyplot as plt
import cv2
def find_image(image):
    fig=plt.figure()
    fig.add_subplot(221)
    print("deep")
    plt.imshow(image)
    gray=cv2.cvtColor(image,cv2.COLOR_BGR2GRAY)
    print("mani")
   
    #converting the imge to gray
    gray=cv2.GaussianBlur(gray,(5,5),0)
    fig=plt.figure()
    fig.add_subplot(222)
    plt.imshow(gray)
    # Gaussian blur is used for removing the noises i.e high frequency values
    # (5,5) is the kernel size.more the kernel value,higher the time to calculate the output
    edged=cv2.Canny(gray,0,225)
    fig.add_subplot(223)
    plt.imshow(edged)
    # 35 -the min value
    # 125 -max value
    cnts=cv2.findContours(edged.copy(),cv2.RETR_LIST,cv2.CHAIN_APPROX_NONE)
    # WE need to use a edged copy as this operation changes the values in edged
    #cv2.RETR_LIST it retrives all the contours
    #cv2.CHAIN_APPROX_SIMPLE  stores only the corner points.
    cnts=imutils.grab_contours(cnts)
    c=max(cnts,key=cv2.contourArea)
    return cv2.minAreaRect(c)

def dist_to_cam(known_width,fl,perwidth):
    return (known_width*fl)/perwidth

At first we need to find focal length of webcam

so we take any rectangle object and place infront of webcam.we calculate the actual width of object and actual distance of object from webcam

focal length=(found width*actual distance)/actual width

now we find distance by (focal length*actual width)/found width
